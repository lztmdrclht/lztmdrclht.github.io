一. 定义：

可以将一个字符串压缩为一个数，在O(1)O(1)O(1)的时间内查找，加快查找效率，降低空间负担。

二. 思想：

    1. 存储:
      将字符串sss看做一个ppp进制数，将其转换为十进制，hash[i]hash[i]hash[i]表示前iii个字符。只需将第iii位数乘以basei−1base^{i-1}basei−1的结果相加，（可以用unsigned long long处理溢出，表示结果模2642^{64}264。用数组ppp预处理basekbase^kbasek【basebasebase常取131131131或133311333113331】）
    1. 查询：
      查询第iii位到第jjj位字符组成的字符串。对比

hash[j]hash[j]hash[j]=s[j]∗p0s[j]*p^{0}s[j]∗p0+s[j−1]∗p1s[j-1]*p^{1}s[j−1]∗p1+...+s[1]∗pj−1s[1]*p^{j-1}s[1]∗pj−1

hash[i−1]hash[i-1]hash[i−1]=s[i−1]∗p0s[i-1]*p^{0}s[i−1]∗p0+s[i−2]∗p1s[i-2]*p^{1}s[i−2]∗p1+...+s[1]∗pi−2s[1]*p^{i-2}s[1]∗pi−2

我们会发现同一位字符hash[i−1]hash[i-1]hash[i−1]比ash[j]ash[j]ash[j]少乘pj−i+1p^{j - i + 1}pj−i+1,所以可以将所求表示为:hash[j] - hash[i-1] * p[j-i+1]，可以理解为将hash[i−1]hash[i-1]hash[i−1]向左移j−i+1j-i+1j−i+1位。

三. 例题：兔子与兔子

    1. 题意：
      mmm次查询 字符串sss中l1l1l1位到第r1r1r1位与l2l2l2位到第r2r2r2是否相同。
    1. 思路：若挨位比较，显然会TTT掉，所以，就可以用上hashhashhash了。存储字符串，每次查询就只需O(1)O(1)O(1)了。真是妙极了。
    1. 代码
#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;
const long long mod = 1e9 + 7, MAXN = 705;
char s[MAXN];
int len, p[MAXN], t[MAXN];
long long dp[MAXN][MAXN][4][4], ans;
void dfs(int l, int r) {
    if(l + 1 == r) {
        dp[l][r][0][1] = 1;
        dp[l][r][1][0] = 1;
        dp[l][r][2][0] = 1;
        dp[l][r][0][2] = 1;
        return ;
    }
    if(p[r] == l) {
        dfs(l + 1, r - 1);
        for(int i = 0; i < 3; i++) {
            for(int j = 0; j < 3; j++) {
                if(i != 1) dp[l][r][1][0] = (dp[l][r][1][0] + dp[l+1][r-1][i][j]) % mod;
                if(j != 1) dp[l][r][0][1] = (dp[l][r][0][1] + dp[l+1][r-1][i][j]) % mod;
                if(i != 2) dp[l][r][2][0] = (dp[l][r][2][0] + dp[l+1][r-1][i][j]) % mod;
                if(j != 2) dp[l][r][0][2] = (dp[l][r][0][2] + dp[l+1][r-1][i][j]) % mod;
            }
        }
    }
    else {
        int x = p[l];
        dfs(l, x);
        dfs(x + 1, r);
        for(int i = 0; i < 3; i++) 
            for(int j = 0; j < 3; j++) 
                for(int u = 0; u < 3; u++) 
                    for(int v = 0; v < 3; v++) 
                        if(!(u == 1 && v == 1 || u == 2 && v == 2))
						 	dp[l][r][i][j] = (dp[l][r][i][j] + (dp[l][x][i][u] * dp[x + 1][r][v][j]) % mod) % mod;
    }
}
int main() {
    scanf("%s", s + 1);
    len = strlen(s + 1);
	for(int i = 1, k = 0; i <= len; i++) {
        if(s[i] == '(') t[k++] = i;
        else {
            p[i] = t[k - 1];
            p[t[--k]] = i;
        }
    }
    dfs(1, len);
	for(int i = 0; i < 3; i++) 
        for(int j = 0; j < 3; j++) 
            ans = (ans + dp[1][len][i][j]) % mod;
    printf("%lld\n", ans);
    return 0;
}
