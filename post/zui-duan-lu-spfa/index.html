最短路——SPFA

II.可有负权边

二. 算法

  • 1. 思想：

    用队列用来保存待处理的结点,只在出队时用队首节点当前的最短路更新所有相邻节点的最短路，出队后不用记录(不同于Dijkstra)，如果到iii节点的最短路更新了，那么与iii相连的节点都需要更新，所以将iii入队。【可以通过记录节点入队次数判断是否有负环】（本质就是暴力）

  • 2. 实现

    I.用一个队列用来保存待处理的结点。

    II.处理时每次取出队首结点uuu，用uuu点当前的最短路长度uuu相连的结点vvv的最短路长度更新，如果vvv点的最短路长度变短，且vvv点不在队列中，就将vvv点放入队尾。

    III.不断从队列中取出结点来进行处理，直至队列空为止。

  • 3. 代码
#include <cstdio>
#include <iostream>
#include <queue>
#include <vector>
#include <algorithm>
using namespace std;
struct node {
	int to, w;
	node(){}
	node(int TO, int W) {
		to = TO;
		w = W;
	}
};
vector<node> G[500005];
queue<int> q;
int n, m, a[100005];
bool f[100005];
void SPFA() {
	for(int i = 2; i <= n; ++i) a[i] = 1e18;
	f[1] = 1;
	q.push(1);
	while(!q.empty()) {
		long long k = q.front();
		q.pop();
		f[k] = 0;
		for(int i = 0; i < G[k].size(); ++i) {
			node t = G[k][i];
			if(a[t.to] > a[k] + t.w) {
				a[t.to] = a[k] + t.w;
				if(!f[t.to]) {
					q.push(t.to); 
					f[t.to] = 1;
				}
			}
		}
	} 
}
int main(){
	scanf("%d %d", &n, &m);
	for(int i = 1, u, v, qz; i <= m; ++i) {
		scanf("%d %d %d", &u, &v, &qz);
		if(u == v) continue;
		G[u].push_back(node(v, qz));
		G[v].push_back(node(u, qz));
	} 
	SPFA();
	printf("%lld\n", a[n]);
    return 0;
}
