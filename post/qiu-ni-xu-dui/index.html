求逆序对

分析

1.考点

归并排序

2.思路

我们想象序列分成左右两个有序集合l[n]l[n]l[n], r[m]r[m]r[m]，若l[i]>r[j]l[i]>r[j]l[i]>r[j],则集合lll中的第iii个到第nnn个元素均大于r[j]r[j]r[j]，分别与r[j]r[j]r[j]组成逆序对。(lll在rrr左边，所以lll中的每个元素的下标都小于rrr中任意一个元素)

回想最初，序列中每个数分别为一个集合。每次便利将其中每两个长kkk的有序集合合并为一个长2k2k2k的有序集合，时间复杂度为O(nlogn),足以搞定1e51e51e5的数据（每次便利扫描n个元素，共log2log_2log2​n次）

代码

#include <cstdio>
#include <iostream>
using namespace std;
const int MAXN = 1000005;
long long n, a[MAXN], b[MAXN], ans;
void gbpx(int l, int r) {
    if (l == r) return;
    int mid = (l + r) / 2, i = l, j = mid + 1, k = l;
    gbpx(l, mid);
    gbpx(mid + 1, r);
    while (i <= mid && j <= r) {
        if (a[i] <= a[j]) b[k++] = a[i++];
        else {
            b[k++] = a[j++];
            ans += mid - i + 1;
        }
    }
    while (i <= mid) b[k++] = a[i++];
    while (j <= r) b[k++] = a[j++];
    for (int u = l; u <= r; u++) a[u] = b[u];
    return;
}
int main() {
    scanf("%d", &n);
    for (int i = 1; i <= n; i++) scanf("%d", &a[i]);
    gbpx(1, n);
    printf("%lld\n", ans);
    return 0;
}
