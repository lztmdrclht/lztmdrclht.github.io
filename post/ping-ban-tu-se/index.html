1.考点

拓扑排序，搜索

2.思路

在涂一个矩形时，能够保证它上面紧挨的所有矩形都已涂过。显然———拓扑排序。

再看数据：n<16n < 16n<16,哟西，爆搜所有拓扑序列，找到次数最小的。

先记录每个矩形上面紧挨的所有矩形，再开始搜索。总共搜到nnn个时结束，更新。每涂一个便枚举所有没有涂同时可以涂的。再来个最优性剪枝，已经大于ansansans了就returnreturnreturn掉。光速AC
。

代码

#include <cstdio>
#include <iostream>
#include <vector>
using namespace std;
struct jx {
	int x1, y1, x2, y2, color;
} p[17];
vector<int> up[17];
int n, ans = 1000005;
bool vis[17];
bool check(int w) { 
	int t = up[w].size();
	for(int i = 0; i < t; i++)  
		if(!vis[up[w][i]])
			return false;
	return true;
}
void dfs(int cnt, int sum, int col) { 
	if(cnt == n) { 
		ans = min(ans, sum);
		return;
	}
	if(sum >= ans)return;
	for(int i = 1; i <= n; i++) {
		if(!vis[i] && check(i)) {
			vis[i] = true;
			dfs(cnt + 1, sum + (col != p[i].color), p[i].color);
			vis[i] = false;
		}
	}
}
int main() {
	scanf("%d", &n);
	ans = n;
	for(int i = 1; i <= n; i++) scanf("%d %d %d %d %d", &p[i].x1, &p[i].y1, &p[i].x2, &p[i].y2, &p[i].color);
	for(int i = 1; i <= n; i++) 
		for(int j = 1; j <= n; j++) 
			if(i != j && p[i].x1 == p[j].x2 && p[i].y2 >= p[j].y1 && p[i].y1 <= p[j].y2)
				up[i].push_back(j);
	dfs(0, 0, 0);
	printf("%d\n", ans);
	return 0;
}
