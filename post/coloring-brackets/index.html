分析

1.考点

区间dp

2.思路情况
  1. 定义状态:dp[l][r][u][v]dp[l][r][u][v]dp[l][r][u][v]表示从lll到rrr的一个合法括号序列,左端颜色为uuu，右端颜色为vvv的染色方案数【uuu，vvv影响两个序列合并】【s[l]s[l]s[l]必然为‘(’，s[r]s[r]s[r]必然为‘）’，否则不合法】。那么，让我们先将 垃圾 情况分类：(大写字母表示一个合法的括号序列)
  • 情况1. l+1=rl + 1 = rl+1=r, 则该序列为： ()
  • 情况2. lll与rrr组成一对括号，则该序列为： (A)(A)(A)
  • 情况3. lll与rrr属于两个不同的合法序列。则该序列为： (A)(B)

  1. 顺序：一个合法的括号序列需要拆分成更短的括号序列，而序列无法直接枚举长度，所以可以通过搜索递归。

  2. 方程

  • 情况1. 边界：(显然)
dp[l][r][1][0] = dp[l][r][2][0] = dp[l][r][0][1] = dp[l][r][0][2] = 1;
  • 情况2. 枚举最外层括号内两端颜色，可得：
if(i != 1) dp[l][r][1][0] = (dp[l][r][1][0] + dp[l + 1][r - 1][i][j]) % mod;
if(j != 1) dp[l][r][0][1] = (dp[l][r][0][1] + dp[l + 1][r - 1][i][j]) % mod;
if(i != 2) dp[l][r][2][0] = (dp[l][r][2][0] + dp[l + 1][r - 1][i][j]) % mod;
if(j != 2) dp[l][r][0][2] = (dp[l][r][0][2] + dp[l + 1][r - 1][i][j]) % mod;
  • 情况3.即将[l,r][l,r][l,r]拆成左右两个合法序列。根据其匹配的括号将序列拆分
dp[l][r][i][j] = (dp[l][r][i][j] + (dp[l][x][i][u] * dp[x + 1][r][v][j]) % mod) % mod;

代码

#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;
const long long mod = 1e9 + 7, MAXN = 705;
char s[MAXN];
int len, p[MAXN], t[MAXN];
long long dp[MAXN][MAXN][4][4], ans;
void dfs(int l, int r) {
    if(l + 1 == r) {
        dp[l][r][0][1] = 1;
        dp[l][r][1][0] = 1;
        dp[l][r][2][0] = 1;
        dp[l][r][0][2] = 1;
        return ;
    }
    if(p[r] == l) {
        dfs(l + 1, r - 1);
        for(int i = 0; i < 3; i++) {
            for(int j = 0; j < 3; j++) {
                if(i != 1) dp[l][r][1][0] = (dp[l][r][1][0] + dp[l+1][r-1][i][j]) % mod;
                if(j != 1) dp[l][r][0][1] = (dp[l][r][0][1] + dp[l+1][r-1][i][j]) % mod;
                if(i != 2) dp[l][r][2][0] = (dp[l][r][2][0] + dp[l+1][r-1][i][j]) % mod;
                if(j != 2) dp[l][r][0][2] = (dp[l][r][0][2] + dp[l+1][r-1][i][j]) % mod;
            }
        }
    }
    else {
        int x = p[l];
        dfs(l, x);
        dfs(x + 1, r);
        for(int i = 0; i < 3; i++) 
            for(int j = 0; j < 3; j++) 
                for(int u = 0; u < 3; u++) 
                    for(int v = 0; v < 3; v++) 
                        if(!(u == 1 && v == 1 || u == 2 && v == 2))
						 	dp[l][r][i][j] = (dp[l][r][i][j] + (dp[l][x][i][u] * dp[x + 1][r][v][j]) % mod) % mod;
    }
}
int main() {
    scanf("%s", s + 1);
    len = strlen(s + 1);
	for(int i = 1, k = 0; i <= len; i++) {
        if(s[i] == '(') t[k++] = i;
        else {
            p[i] = t[k - 1];
            p[t[--k]] = i;
        }
    }
    dfs(1, len);
	for(int i = 0; i < 3; i++) 
        for(int j = 0; j < 3; j++) 
            ans = (ans + dp[1][len][i][j]) % mod;
    printf("%lld\n", ans);
    return 0;
}
